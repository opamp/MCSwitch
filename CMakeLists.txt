PROJECT(MCSwitch)
CMAKE_MINIMUM_REQUIRED(VERSION 2.8.10.2)

SET(MCSWITCH_VERSION_MAJOR "0")
SET(MCSWITCH_VERSION_MINOR "4")
SET(MCSWITCH_VERSION_PATCH "1")
SET(MCSWITCH_VERSION "${MCSWITCH_VERSION_MAJOR}.${MCSWITCH_VERSION_MINOR}.${MCSWITCH_VERSION_PATCH}")
MESSAGE(STATUS "MCSWITCH VERSION -> ${MCSWITCH_VERSION}")


SET(SRC_DIR ${PROJECT_SOURCE_DIR}/src)

SET(P_LINUX OFF)
SET(P_OSX OFF)

IF(NOT APPLE AND NOT UNIX)
    MESSAGE(WARNING "Your Platform isn't supported!!") #cmake warn user,but continue.
ELSEIF(APPLE)
    SET(P_LINUX OFF)
    SET(P_OSX ON)
ELSEIF(UNIX AND NOT APPLE)
    SET(P_LINUX ON)
    SET(P_OSX OFF)
ENDIF(NOT APPLE AND NOT UNIX)


CONFIGURE_FILE(${PROJECT_SOURCE_DIR}/config.hpp.in ${CMAKE_CURRENT_BINARY_DIR}/config.hpp)

INCLUDE(${SRC_DIR}/CMakeLists.txt)
